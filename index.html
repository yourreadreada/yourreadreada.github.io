Student Management System

A comprehensive Python-based system to manage student records with efficient data structures. The system maintains student information using dictionaries and tuples, allowing users to add records, filter students by performance (average marks > 80), and update marks dynamically. Features a user-friendly menu-driven interface for easy interaction.

Python Data Structures Dictionary & Tuple List Management
View Full Python Code 
# Student Record Manager - Complete Implementation

students = {}  # Declaring Empty Dictionary To Store Values

n = int(input("Enter Number Of Students: "))  # Taking Initial Student Record

for i in range(n):  # Loop To Input Initial Student Record
    print()
    name = input("Enter Name: ")
    roll = int(input("Enter Roll Number: "))
    limit = int(input("Enter Number Of Marks To Be Entered: "))

    marks = []  # Creating Empty List To Store Marks For A Student

    for j in range(limit):  # Loop To Store Marks For A Student
        inputmark = int(input(f"Enter Mark {j+1}: "))
        marks.append(inputmark)  # Adding Marks To Marks List

    students[roll] = (name, marks)  # Storing Student Record As Tuple

print("\n All Students:")
print(students)  # Printing Initial Entered Students

while True:
    print("\n Welcome To Student Record Manager:")
    print("1. Add A Student Record")
    print("2. Display Students With Average Marks Above 80")
    print("3. Update Marks For A Specific Roll Number")
    print("4. Exit")

    choice = int(input("\n Enter A Choice: "))  # Taking Choice Input

    if choice == 1:  # Adding Student Record
        print()
        name = input("Enter Name: ")
        roll = int(input("Enter Roll Number: "))
        limit = int(input("Enter Number Of Marks To Be Entered: "))

        marks = []
        for j in range(limit):
            inputmark = int(input(f"Enter Mark {j+1}: "))
            marks.append(inputmark)

        students[roll] = (name, marks)
        print(f"Student {name} Added Successfully.")

    elif choice == 2:  # Displaying Students With Average Marks Above 80
        print("\nStudents With Average Marks Above 80:")
        found = False  # To Check If Any Student Is Found

        for roll, (name, marks) in students.items():
            total = 0
            for mark in marks:  # Loop To Calculate Average
                total += mark
            average = total / len(marks)

            if average > 80:  # If Average > 80
                print(f"{name} (Roll: {roll}) - Average: {average:.2f}")
                found = True
        
        if not found:
            print("No Students Found With Average Above 80.")

    elif choice == 3:  # Update Marks For A Specific Roll Number
        roll = int(input("\nEnter Roll Number to update marks: "))

        if roll in students:  # Checking If Roll Number Exists
            name, marks = students[roll]
            print(f"Current Marks For {name}: {marks}")

            limit = int(input("Enter Number Of New Marks: "))
            newmarks = []  # Creating New Marks List

            for j in range(limit):  # Loop To Input New Marks
                inputmark = int(input(f"Enter Mark {j+1}: "))
                newmarks.append(inputmark)

            students[roll] = (name, newmarks)  # Updating Dictionary
            print(f"Marks Updated Successfully For {name}!")

        else:
            print("Roll Number Not Found.")

    elif choice == 4:  # Exit Program
        print("Exiting Program.")
        break  # Exiting While Loop

    else:  # For Invalid Choice
        print("Invalid Choice.")

Grocery Billing System

An interactive Python application for managing grocery store operations. The system maintains stock inventory using dictionaries and processes customer purchases with tuple-based bill tracking. It validates stock availability, updates inventory in real-time, and generates professional itemized bills with total calculations.

Python Data Structures Inventory Management Bill Processing
View Full Python Code 
# Grocery Billing System - Complete Implementation

stock = {}  # Declaring Empty Dictionary To Store Available Stock
billitems = []  # Declaring Empty List To Store Bill Values As Tuples

print(" Welcome to Grocery Billing System:- ")
n = int(input("Enter Number Of Items To Add To Stock: "))

for i in range(n):  # Loop To Input Initial Stock
    print(f"\n Stock Item {i+1} : ")
    itemname = input("Enter Item Name: ").lower()
    quantity = int(input("Enter Item Quantity: "))
    unitprice = float(input("Enter Price Per Unit: "))
    stock[itemname] = {'quantity': quantity, 'unitprice': unitprice}

# Display Initial Stock
print("\n Initial Stock:")
for item, info in stock.items():
    print(f"{item.capitalize()}: {info['quantity']} Units at Rs.{info['unitprice']:.2f} per unit")

# Menu Loop
while True:
    print("\n Grocery Billing Menu: ")
    print("1. Purchase Items")
    print("2. Show Current Stock")
    print("3. Print Final Bill")
    print("4. Exit")

    choice = int(input("\n Enter A Choice: "))  # Taking Choice As Input

    if choice == 1:  # Purchasing Items
        purchaseitem = int(input("Enter Number Of Items To Purchase: "))
        for i in range(purchaseitem):
            print(f"\n Purchase Item {i+1} : ")
            itemname = input("Enter Item Name: ").lower()

            if itemname not in stock:  # Check if item exists
                print(f" Item '{itemname}' NOT Available.")
                continue

            quantity = int(input("Enter Quantity: "))
            available_qty = stock[itemname]['quantity']
            unitprice = stock[itemname]['unitprice']

            if available_qty >= quantity:  # Check stock availability
                billitems.append((itemname, quantity, unitprice))
                stock[itemname]['quantity'] -= quantity
                print(f" {itemname.capitalize()} Added To Bill. Remaining: {stock[itemname]['quantity']}")
            else:
                print(f" Insufficient Stock! Only {available_qty} Units Available.")

    elif choice == 2:  # Showing Current Stock
        print("\n Current Stock Levels:")
        for item, info in stock.items():
            print(f"{item.capitalize()}: {info['quantity']} Units at Rs.{info['unitprice']:.2f}")

    elif choice == 3:  # Printing Final Bill
        if not billitems:
            print("\n No Items Purchased Yet.")
        else:
            print("\n" + "=" * 50)
            print("           FINAL BILL")
            print("=" * 50)
            totalbill = 0
            for itemname, quantity, unitprice in billitems:
                totalitem = quantity * unitprice
                totalbill += totalitem
                print(f"{itemname.capitalize()}: {quantity} x Rs.{unitprice:.2f} = Rs.{totalitem:.2f}")
            print("-" * 50)
            print(f"TOTAL AMOUNT: Rs.{totalbill:.2f}")
            print("=" * 50)
            print("\nThank You For Shopping!")

    elif choice == 4:  # Exit Program
        print("Exiting Grocery Billing System.")
        break

    else:  # Invalid Choice
        print("Invalid choice.")
